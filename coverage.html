package hofs

// Every is a generic function that accepts a slice of type T and a function f
// that accepts a single argument of type T and returns a boolean value.
// The function returns true if all elements in the slice satisfy the
// condition specified by the f function, and false otherwise.
func Every[T any](slice []T, f func(arg T) bool) bool {
        every := true

        for _, item := range slice {
                ok := f(item)

                if !ok {
                        every = false
                        break
                }
        }

        return every
}

// Filter will receive a slice of T and return a slice of the T elements
// that return true in the function passed as 2nd argument.
func Filter[T any](slice []T, f func(arg T) bool) []T {
        var filteredSlice []T

        for _, item := range slice {
                if f(item) {
                        filteredSlice = append(filteredSlice, item)
                }
        }

        return filteredSlice
}

// Find receives a slice of type T and returns the first T element that returns true
// in the function passed as the 2nd argument.
func Find[T any](slice []T, f func(arg T) bool) *T {
        for _, item := range slice {
                ok := f(item)
                if ok {
                        return &item
                }
        }

        return nil
}

// Map will receive a slice of T values and return a slice of Y values of the same length.
func Map[T any, Y any](slice []T, f func(arg T) Y) []Y {
        result := make([]Y, 0, len(slice))
        for _, item := range slice {
                newItem := f(item)
                result = append(result, newItem)
        }

        return result
}

// Some is a generic function that accepts a slice of type T and a function f
// that accepts a single argument of type T and returns a boolean value.
// Some returns true if at least one element in the slice satisfies
// the condition specified by the f function, and false otherwise.
func Some[T any](slice []T, f func(arg T) bool) bool {
        some := false

        for _, item := range slice {
                ok := f(item)
                if ok {
                        some = true
                        break
                }
        }

        return some
}
